		name = news.175.c
		trigger = {
			OR = {
				tag = GER

			}
		}
	}

	#Britain has thrown in their lot.
	option = {
		name = news.175.e
		trigger = {
			FRA = {
				has_war = yes
			}
			NOT = {
				has_war_with = FRA
			}
			NOT = {
				OR = {
					tag = FRA

		}
	}
}
