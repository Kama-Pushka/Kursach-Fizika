country_event = {
	id = NWpalestine.3
	title = NWpalestine.3.t
	desc = NWpalestine.3.desc
	picture = GFX_report_event_bpl_3
	fire_only_once = yes
	
	immediate = { log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): event NWpalestine.3" }
	immediate = {
		set_country_flag = BPL_arab_declare_indepandance_flag
		hidden_effect = {
			every_state = {
				limit = {
					OR = {
						state = 454
						state = 978
					}
				}
				PAL = {
					transfer_state = PREV
					add_state_core = PREV
				}
			}
			if = {
				limit = { PAL = { num_divisions < 1 } }
				PAL = {
					if = {
						limit = { NOT = { has_template = "Arab Divisions" } }
						division_template = {
							name = "Arab Divisions"
							is_locked = no
							regiments = {
								infantry = { x = 0 y = 0 }
								infantry = { x = 0 y = 1 }
								infantry = { x = 0 y = 2 }
								infantry = { x = 1 y = 0 }
								infantry = { x = 1 y = 1 }
								infantry = { x = 1 y = 2 }
							}
						}
					}
					random_owned_controlled_state = {
						create_unit = {
							division = "name = \"Arab Divisions\" division_template = \"Arab Divisions\" start_experience_factor = 0.2" 
							owner = PREV
						}
						create_unit = {
							division = "name = \"Arab Divisions\" division_template = \"Arab Divisions\" start_experience_factor = 0.2" 
							owner = PREV
						}
						create_unit = {
							division = "name = \"Arab Divisions\" division_template = \"Arab Divisions\" start_experience_factor = 0.2" 
							owner = PREV
						}
					}
				}
			}
			clear_variable = BPL_arab_tensions_total
			clear_variable = BPL_jews_tensions_total
			set_country_flag = BPL_jewish_arab_confrontation_flag
		}
	}
	
	mean_time_to_happen = { days = 7 }
	
	trigger = {
		tag = BPL
		NOT = {
			has_country_flag = BPL_arab_declare_indepandance_flag
			has_completed_focus = BPL_concessions_arab_leaders
		}
		check_variable = { BPL_arab_tensions_total > 99.99 }
		NOT = { has_country_flag = BPL_jewish_arab_confrontation_flag }
	}
	
	option = {
		name = NWpalestine.3.a
		ai_chance = {
			factor = 75
		}
		PAL = {
			declare_war_on = {
				target = ROOT
				type = civil_war
			}
			add_civil_war_target = ROOT
			hidden_effect = { add_ideas = generic_civil_war }
		}
		add_civil_war_target = PAL
		hidden_effect = { add_ideas = generic_civil_war }
		effect_tooltip = { unlock_national_focus = BPL_jewish_arab_compromise }
		effect_tooltip = { unlock_national_focus = BPL_concessions_jewish_leaders }
	}
	
	option = {
		name = NWpalestine.3.b
		ai_chance = {
			factor = 25
		}
		effect_tooltip = { unlock_national_focus = BPL_jewish_arab_compromise }
		effect_tooltip = { unlock_national_focus = BPL_concessions_jewish_leaders }
	}
	
	option = {
		name = NWpalestine.3.c
		trigger = {
			is_ai = no
		}
		PAL = { change_tag_from = ROOT }
		hidden_effect = {
			add_ideas = generic_civil_war
			random_list = {
				75 = {
					PAL = {
						declare_war_on = {
							target = ROOT
							type = civil_war
						}
						add_civil_war_target = ROOT
						add_ideas = generic_civil_war
					}
					add_civil_war_target = PAL
				}
				25 = { }
			}
		}
	}
}

		tag = BPL
		country_exists = GER

		NOT = { has_completed_focus = BPL_stop_jewish_immigration }
	}
	
	option = {
		name = NWpalestine.5.a
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		if = {
			limit = {
				GER = {
					any_controlled_state = {
						OR = {
							is_core_of = SOV
							is_core_of = POL
						}
					}
				}
			}
			add_manpower = 500
		}
		else = { add_manpower = 200 }
		if = {
			limit = { has_variable = BPL_arab_tensions_total }
			set_temp_variable = { BPL_arab_tensions = 2 }
