# Чернокожие и цветные начали восстание
country_event = {
	id = NW_SAF.3
	title = NW_SAF.3.t
	desc = NW_SAF.3.d
	picture = GFX_report_event_hol_21_26
	is_triggered_only = yes
	trigger = {
		if = {
			limit = { is_ai = yes }
			OR = {
				has_country_flag = NW_SAF_AI_DEMOCRATIC_BLACK_COUP_FLAG
				has_country_flag = NW_SAF_AI_TROZKISM_FLAG
				has_country_flag = NW_SAF_AI_COMMUNIST_FLAG
			}
		}
	}
	immediate = {
		log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): event NW_SAF.3"
		hidden_effect = {
			set_country_flag = SAF_civil_war_flag
			mark_focus_tree_layout_dirty = yes
			SAB = {
				PREV = {
					random_owned_controlled_state = {
						prioritize = { 1108 }
						limit = { is_core_of = PREV }
						transfer_state_to = PREV.PREV
					}
				}
				promote_character = SAF_anc_joint_command
				add_civil_war_target = PREV
				PREV = {
					every_core_state = { add_core_of = PREV.PREV }
					add_civil_war_target = PREV
				}
				declare_war_on = { target = PREV type = civil_war }
				every_enemy_country = {
					every_state = {
						limit = {
							NOT = { is_fully_controlled_by = PREV }
							controller = { tag = PREV.PREV }
						}
						teleport_armies = {
							limit = { tag = PREV }
							to_state_array = controlled_states
						}
					}
				}
				if = {
					limit = { NOT = { has_template = "Opposition Forces" } }
					division_template = {
						name = "Opposition Forces"	
						is_locked = no
						division_names_group = SAF_INF_01
						regiments = {
							infantry = { x = 0 y = 0 }
							infantry = { x = 0 y = 1 }
							infantry = { x = 0 y = 2 }
						}
					}
				}
				if = {
					limit = { NOT = { has_template = "Peoples' Militia" } }
					division_template = {
						name = "Peoples' Militia"	
						is_locked = no
						division_names_group = SAF_INF_01
						regiments = {
							irregular_infantry = { x = 0 y = 0 }
							irregular_infantry = { x = 0 y = 1 }
							irregular_infantry = { x = 0 y = 2 }
						}
					}
				}
				capital_scope = {
					create_unit = {
						division = "division_template = \"Peoples' Militia\" start_experience_factor = 0.0 start_equipment_factor = 0.2"
						owner = PREV
						count = 4
					}
					create_unit = {
						division = "division_template = \"Opposition Forces\" start_experience_factor = 0.1 start_equipment_factor = 0.4"
						owner = PREV
						count = 2
					}
				}
				inherit_technology = PREV
				set_temp_variable = { SAF_num_research_slots_temp = PREV.amount_research_slots }
				set_research_slots = SAF_num_research_slots_temp
				add_research_slot = -1
				set_temp_variable = { ideology = token:trozkism }
				NW_add_to_coalition = yes
				set_temp_variable = { ideology = token:communism }
				NW_add_to_coalition = yes
				mark_focus_tree_layout_dirty = yes
				complete_national_focus = SAF_uprising_against_white_oppression
				SAF_civil_war_check_effect = yes
			}
		}
	}
	option = {
		name = NW_SAF.3.a

	}
	option = {
		name = NW_SAF.3.b
		hidden_effect = {
			if = {
				limit = { num_divisions < 8 }
				set_temp_variable = { num_divisions_temp_0 = num_divisions }
				set_temp_variable = { num_divisions_temp_1 = 8 }
				subtract_from_temp_variable = { num_divisions_temp_1 = num_divisions_temp_0 }
				if = {
					limit = { check_variable = { var = num_divisions_temp_1 value = 0 compare = greater_than_or_equals } }
					if = {
						limit = { NOT = { has_template = "Disctrict Force" } }
						division_template = {
							name = "Disctrict Force"	
							is_locked = no
							division_names_group = SAF_INF_01
							regiments = {
								infantry = { x = 0 y = 0 }
								infantry = { x = 0 y = 1 }
								infantry = { x = 0 y = 2 }
								infantry = { x = 1 y = 0 }
								infantry = { x = 1 y = 1 }
								infantry = { x = 1 y = 2 }
							}
						}
					}
					while_loop_effect = {
						limit = { check_variable = { num_divisions_temp_1 > 0 } }
						add_to_temp_variable = { num_divisions_temp_1 = -1 }
						random_owned_controlled_state = {
							limit = { is_core_of = ROOT }
							create_unit = {
								division = "division_template = \"Disctrict Force\" start_experience_factor = 0.2 start_equipment_factor = 0.5"
								owner = ROOT
								count = 1
							}
						}
					}
				}
			}
		}
		SAB = { change_tag_from = ROOT }
		custom_effect_tooltip = empty_line_tooltip
		unlock_decision_category_tooltip = SAF_fighting_the_tyranny_of_white_supremacy_category
	}
}
