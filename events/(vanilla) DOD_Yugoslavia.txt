			custom_effect_tooltip = DOD_yugoslavia.1.a.tt
			swap_ruler_traits = { remove = underage_monarch add = legitimate_heir }
			
			hidden_effect = {
				YUG_peter_ii = {
					promote_character = { ideology = dual_monarchy }
					promote_character = { ideology = conservatism }
				}

						}
					}

					limit = {
						has_government = monarchy
					}

						character = YUG_peter_ii
						ideology = conservatism
						trait = underage_monarch
					}
					add_trait = {
						character = YUG_peter_ii
						ideology = conservatism
						trait = legitimate_heir
					}
				}
				if = {
					limit = {
						has_government = democratic
					}
					remove_trait = {
						character = YUG_peter_ii
						ideology = dual_monarchy
						trait = underage_monarch
					}
					add_trait = {
						character = YUG_peter_ii
						ideology = dual_monarchy
						trait = legitimate_heir
					}
				}
				set_country_flag = peter_is_of_age
			}
		}
		GER = {
			country_event = DOD_yugoslavia.2
			add_ai_strategy = {
				type = alliance
				id = ROOT		
				value = -1000
			}
		}
		if = {
			limit = {
				is_subject_of = GER
			}
			set_autonomy = {
				target = YUG
				autonomy_state = autonomy_free
			}

			limit = {
				is_in_faction_with = GER

				type = alliance

				type = alliance
				id = "SOV"		
				value = -1000

				type = alliance
				id = "JAP"		
				value = -1000

				value = -1000

				id = "ENG"		
				value = 1000

					type = alliance
					id = ROOT

			factor = 30
			modifier = {
				factor = 0

			}

				109 = { is_owned_by = YUG }
			}

					limit = {
						104 = { is_owned_by = YUG }
					}
					set_capital = { state =  104 }
				}
			}

				}
			}
		}
	}
