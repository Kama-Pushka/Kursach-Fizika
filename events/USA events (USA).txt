	id = NW_USA.1
	title = NW_USA.1.t
	desc = NW_USA.1.d
	picture = GFX_report_event_NW_USA.0
	is_triggered_only = yes
	trigger = { has_country_leader = { character = USA_franklin_delano_roosevelt ruling_only = yes } }
	immediate = { log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): event NW_USA.1" }
	option = {
		name = NW_USA.1.a
		if = {
			limit = {
				FROM = {
					has_completed_focus = JAP_cyclone_matsuoka
					NOT = {
						has_completed_focus = JAP_restore_the_Contract_Fraction
						has_country_flag = JAP_purge_of_Toseiha_members_in_the_Army_country_flag
					}
					check_variable = { JAP_the_level_of_insubordination_var > 0 }
				}
			}
			hidden_effect = { country_event = { id = NW_USA.2 days = 30 } }
		}
		else = {
			FROM = {
				add_opinion_modifier = {
					modifier = USA_private_diplomacy_op
					target = ROOT
				}
				reverse_add_opinion_modifier = {
					target = ROOT
					modifier = USA_private_diplomacy_op
				}
			}
		}
	}
	option = {
		name = NW_USA.1.b
	}
}
