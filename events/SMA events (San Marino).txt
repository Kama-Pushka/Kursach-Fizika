add_namespace = san_marino

#Свержение фашисткой диктатуры
country_event = { 
	id = san_marino.1
	title = san_marino.1.t
	desc = san_marino.1.d
	picture = GFX_report_event_generic_peaceful_annexation
	trigger = { 
		original_tag = SMA
		SMA = {
			has_government = fascism
			is_subject = no
		}
		NOT = {
			any_neighbor_country = { 
				OR = {
					has_government = fascism
					has_government = communism
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 3
	}
	
	immediate = { log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): event san_marino.1" }
	
	option = {
		name = san_marino.1.a
		set_politics = { ruling_party = democratic }
		set_political_party = {
			ideology = democratic
			popularity = 40
		}
	}
}
#Возвращение фашистов
country_event = {
	id = san_marino.2
	title = san_marino.2.t
	desc = san_marino.2.d
	picture = GFX_report_event_generic_peaceful_annexation
	trigger = { 
		original_tag = SMA
		SMA = {
			NOT = {
				has_government = fascism
			}
			is_subject = no
		}
		any_neighbor_country = { 
			has_government = fascism
		}
	}
	
	mean_time_to_happen = {
		days = 3
	}
	
	immediate = { log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): event san_marino.2" }
	
	option = {
		name = san_marino.2.a
		if = {
			limit = {
				NOT = {
					has_country_flag = fascism_return_event
				}
			}
			set_party_name = { 
				ideology = fascism 
				long_name = SMA_fascism2_party_long 
				name = SMA_fascism2_party
			}
		}
		set_politics = { ruling_party = fascism }
		set_political_party = {
			ideology = fascism
			popularity = 40
		}
		set_country_flag = fascism_return_event
	}
}
