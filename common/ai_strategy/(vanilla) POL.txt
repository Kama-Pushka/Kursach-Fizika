	# UK has a mirrored version of this

		original_tag = POL

			has_government = neutrality
			has_government = democratic
		}
		ENG = { has_government = democratic }
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = equipment_market_trade_desire

	}
}
