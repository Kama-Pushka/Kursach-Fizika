on_actions = {
	on_startup = {
		effect = {
			JAP = {
				activate_mission = JAP_session_of_the_ioc_mis
			}
			ENG = {
				country_event = { id = NW_JAP.olympiad.2 days = 180 }
			}
		}
	}
	on_daily_JAP = {
		effect = {
			if = { # TODO че это и для кого оно
				limit = {
					has_country_flag = JAP_olympic_committee_support_for_the_relay_race_flag
					has_country_flag = JAP_ministry_of_internal_affairs_support_for_the_relay_race_flag
					has_country_flag = JAP_international_participants_support_for_the_relay_race_flag
					NOT = { has_country_flag = JAP_all_support_for_the_relay_race_flag }
				}
				set_country_flag = JAP_all_support_for_the_relay_race_flag
				add_to_variable = { JAP_degree_of_readiness_to_olympiad_var = 10 }				
			}
		}
	}
	on_declare_war = {
		effect = { # TODO
			JAP = {
				if = {
					limit = { has_active_mission = JAP_prepare_to_olympiad_mis }
					set_temp_variable = { count_of_major_at_war = 0 }
					every_country = {
						limit = {
							is_major = yes
							has_war_with_major = yes
						}
						add_to_variable = { count_of_major_at_war = 1 }
					}
					if = {
						limit = {
							check_variable = { count_of_major_at_war > 2 }
							NOT = { has_country_flag = JAP_86_news_event_complete_flag }
						}
						news_event = { id = NW_JAP.olympiad.14 hours = 1 random_days = 1 }
						set_country_flag = JAP_86_news_event_complete_flag

						clr_country_flag = JAP_prepare_to_olympiad_flag
						clr_country_flag = JAP_organizing_committee_created_flag
					}
				}
			}
		}
	}
}
