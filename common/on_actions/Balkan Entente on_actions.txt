on_actions = {
	
	on_declare_war = {
   		effect = {
        	if = {
            	limit = {
					FROM = {
						is_in_array = { global.balkan_entente = THIS }
						is_in_faction = no
					}
					ROOT = { is_in_array = { global.country_in_europe = THIS }}
				}
				ROOT = {
					every_other_country = {
						limit = {
							is_in_array = { global.balkan_entente = THIS }
							can_declare_war_on = ROOT
							is_in_faction = no
						}
						country_event = { id = NW_balkan_entente.0 days = 5 random_days = 3 }
					}
				}
			}
		}
	}
	on_assume_faction_leadership = {
   		effect = {
        	if = {
            	limit = {
					FROM = {
						is_in_array = { global.balkan_entente = THIS }
						has_country_flag = NW_leader_of_balkan_military_alliance
					}
					ROOT = {
						is_in_array = { global.balkan_entente = THIS }
						has_global_flag = NW_balkan_military_alliance_created
					}
				}
				FROM = { clr_country_flag = NW_leader_of_balkan_military_alliance }
				ROOT = { set_country_flag = NW_leader_of_balkan_military_alliance }
			}
		}
	}
}
