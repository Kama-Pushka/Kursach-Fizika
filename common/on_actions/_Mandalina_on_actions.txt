on_actions = {
    on_startup = {
        effect = { #Effects Nazi flag
            if = {
                limit = {
                    has_game_rule = { rule = censor_setting option = RULE_CENSORED }
                }
                GER = {
					set_cosmetic_tag = GER_censored
				}
                ICE = { #for ICE_fascism flag
					set_cosmetic_tag = ICE_censored
				}
                DEN = { #for DEN_fascism flag
                    set_cosmetic_tag = DEN_censored
                }
            }
        }
    }

    on_weekly_GER = { #Just in case the flag changes
        effect = {
            if = {
                limit = {
                    has_game_rule = { rule = censor_setting option = RULE_CENSORED }
                    NOT = {
                        OR = {
                            has_cosmetic_tag = GER_censored
                            has_cosmetic_tag = GER_cosmetic #Enter all cosmetic tags used in Germany
                        }
                    }
                }
                GER = {
					set_cosmetic_tag = GER_censored
				}
            }
        }
    }

}
