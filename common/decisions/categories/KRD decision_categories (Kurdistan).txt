			}

				}
				has_country_flag = NWkurd3_flag
			}
			NOT = { has_country_flag = kurd_revolt_flag }
			NOT = { has_country_flag = kurd_revolt_over_flag }
		}
	}
	visible = {
		OR = {
			AND = {
				any_owned_state = {
					NOT = { is_core_of = ROOT }
					is_core_of = KRD
				}
				has_country_flag = NWkurd1_a_flag
			}
			has_country_flag = NWkurd3_flag
		}
		NOT = { has_country_flag = kurd_revolt_flag }
		NOT = { has_country_flag = kurd_revolt_over_flag }
