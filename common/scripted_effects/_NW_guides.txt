# character, ideology, start_focus
NW_guides_add_character = {
	add_to_variable = { global.NW_guides_character_scope_count = 1 }
	if = {
		limit = { check_variable = { global.NW_guides_character_scope_count = global.states^num } }
		set_variable = { global.NW_guides_character_scope_count = 1 }
		log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): обнуление счетчика (NW_guides_add_character)"
	}

	add_to_array = { NW_guide_character_array = global.NW_guides_character_scope_count }
	var:global.NW_guides_character_scope_count = {
		set_variable = { NW_character_token@ROOT = character }
		set_variable = { NW_ideology_token@ROOT = ideology }
		set_variable = { NW_focus_token@ROOT = start_focus }
	}
	
	NW_guides_set_x_y_position = yes
}

# x_frame, y_frame
NW_guides_set_x_y_position = {
	# clamp_temp_variable = { var = x_frame min = 0 max = 4 } # макс 5 лидеров в линию
	
	set_temp_variable = { x = 120 }
	set_temp_variable = { y = 180 }
	multiply_temp_variable = { x = x_frame }
	multiply_temp_variable = { y = y_frame }
	add_to_temp_variable = { x = 15 } # стартовая позиция leader
	add_to_temp_variable = { y = 8 } # стартовая позиция leader
	add_to_array = { _NW_guides_l_x = x }
	add_to_array = { _NW_guides_l_y = y }
	# log = "CHECK [?x] [?y]"
	
	set_temp_variable = { x = 120 }
	set_temp_variable = { y = 180 }
	multiply_temp_variable = { x = x_frame }
	multiply_temp_variable = { y = y_frame }
	add_to_temp_variable = { x = 10 } # стартовая позиция leader_frame2
	add_to_temp_variable = { y = 0 } # стартовая позиция leader_frame2
	add_to_array = { _NW_guides_lf_x = x }
	add_to_array = { _NW_guides_lf_y = y }
	
	set_temp_variable = { x = 120 }
	set_temp_variable = { y = 180 }
	multiply_temp_variable = { x = x_frame }
	multiply_temp_variable = { y = y_frame }
	add_to_temp_variable = { x = -18 } # стартовая позиция leader_frame2
	add_to_temp_variable = { y = 122 } # стартовая позиция leader_frame2
	add_to_array = { _NW_guides_ln_x = x }
	add_to_array = { _NW_guides_ln_y = y }
}

# character
NW_guides_get_character_scope_for_token = {
	for_each_scope_loop = {
		array = NW_guide_character_array
		break = b
		
		if = {
			limit = {
				check_variable = { NW_character_token@ROOT = character }
			}
			set_temp_variable = { character = THIS.id }
			set_temp_variable = { b = 1 }
		}
	}
}

# root_character, swap_character
NW_guides_set_swap_character_data = {
	var:root_character = {
		set_variable = { NW_swap_character@ROOT = swap_character }
		# clr_state_flag = NW_guides_character_hidden@ROOT
	}
	var:swap_character = {
		set_state_flag = NW_guides_character_hidden@ROOT
	}
}

# character
NW_guides_set_first_in_list = {
	var:character = {
		clr_state_flag = NW_guides_character_hidden@ROOT
	}
}
