		limit = {

		}
		country_event = NW_JAP_aht.14
	}
}

JAP_aht_fractions_influence_update = {
	set_temp_variable = { loyal = 0 }
	set_temp_variable = { opposition = 0 }
	for_each_scope_loop = {
		array = JAP_aht_fractions
		
		clamp_variable = { var = JAP_aht_influence min = 0 }
		if = {
			limit = {
				check_variable = { JAP_aht_fraction_position = 1 }
			}
			add_to_temp_variable = { loyal = JAP_aht_influence }
		}
		else_if = {
			limit = {
				check_variable = { JAP_aht_fraction_position = 3 }
			}
			add_to_temp_variable = { opposition = JAP_aht_influence }
		}
	}
	set_variable = { JAP_aht_fraction_support = loyal }
	add_to_temp_variable = { opposition = loyal }
	divide_variable = { JAP_aht_fraction_support = opposition } # TODO opposition=0 ?
}

JAP_aht_faction_status_update = {
	clamp_temp_variable = { var = position min = 1 max = 3 }

		set_variable = { var:faction:JAP_aht_fraction_position = position }
	}
	JAP_aht_fractions_influence_update = yes # TODO вынести наружу?
}
