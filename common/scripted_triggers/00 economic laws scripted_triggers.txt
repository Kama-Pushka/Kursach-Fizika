		always = no
	}
	else_if = {
		limit = {
			original_tag = TUR
			OR = {
				has_completed_focus = TUR_the_clodius_agreement_2
				has_completed_focus = TUR_the_clodius_agreement_3
			}
		}
		OR = {
			is_subject = yes

	}
