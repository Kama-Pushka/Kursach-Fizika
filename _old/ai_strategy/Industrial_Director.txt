industrialization_effort = {
	enable = {
		threat < 0.2
		NOT = {
			OR = {
				has_war = yes
				late_game_nation = yes
				is_historical_focus_on = yes
			}
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = added_military_to_civilian_factory_ratio
		value = -100
	}
	ai_strategy = {
		type = dockyard_to_military_factory_ratio
		value = -100
	}
	ai_strategy = {
		type = building_target	
		id = industrial_complex
		value = 9999
	}

	ai_strategy = {
		type = building_target
		id = fuel_silo
		value = -9999
	}

	ai_strategy = {
		type = equipment_market_spend_factories
		value = -9999  # Factor for the EQUIPMENT_MARKET_MAX_CIVS_FOR_PURCHASES_RATIO define (value = 20 results in a factor of 1.2)	
	}
}

we_need_a_bit_of_rubber = {
	enable = {
		has_war = yes
		num_of_civilian_factories > 50
		num_of_military_factories > 50
		NOT = {
			OR = {
				late_game_nation = yes
				dont_need_any_more_rubber = yes
			}
		}
		check_variable = { resource_consumed@rubber > resource_produced@rubber }
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = building_target
		id = synthetic_refinery
		value = 9999
	}
}
